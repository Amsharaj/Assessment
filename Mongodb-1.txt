            from: "personal",
            localField: "_id",
            foreignField: "_id",
            as: "personal_details"

    },

        $project: {
            _id: 1,
            emp_name: 1,
            salary: 1,
            role: 1,
            personal_details: { $arrayElemAt: ["$personal_details", 0] }
